叫醒eth0
# ifconfig eth0 up
# dhclient eth0

安裝network manager
# apt-get update
# apt-get install network-manager-gnome

啟用dynamic dns update
參考連結
http://askubuntu.com/questions/137037/networkmanager-not-populating-resolv-conf
# dpkg-reconfigure resolvconf

修正時區
# dpkg-reconfigure tzdata

去ntp服務器抓取正確時間(一次性)
# ntpdate ntp.ubuntu.com

設定cron，常態性更新系統時間
root@linaro-alip:/etc/cron.daily# cd /root
root@linaro-alip:~# touch /etc/cron.daily/ntpdate
root@linaro-alip:~# chmod 755 /etc/cron.daily/ntpdate 
root@linaro-alip:~# echo '#!/bin/sh' >> /etc/cron.daily/ntpdate 
root@linaro-alip:~# echo 'ntpdate ntp.ubuntu.com' >> /etc/cron.daily/ntpdate
root@linaro-alip:~# cat /etc/cron.d
cron.d/     cron.daily/ 
root@linaro-alip:~# cat /etc/cron.daily/ntpdate 
#!/bin/sh
ntpdate ntp.ubuntu.com
root@linaro-alip:~# 


安裝vim (修正上下左右變ABCD)
# apt-get install vim

設定hostname
修改/etc/hostname
原來的內容是
linaro-alip
改成
rose01

修改/etc/hosts
一樣把原來的主機名
127.0.1.1       linaro-alip
改成
127.0.1.1       rose01

修正方式
修改/etc/default/rcS
(把原來的#FSCKFIX=no改成FSCKFIX=yes)
root@rose01:~# cat /etc/default/rcS | grep FSCKFIX
#FSCKFIX=no
root@rose01:~# vi /etc/default/rcS 
root@rose01:~# cat /etc/default/rcS | grep FSCKFIX
FSCKFIX=yes
